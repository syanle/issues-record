[gae]
enable = 1
appid = goagent
password = 这块需要配置密码（防止被盗）

[iplist]
google_cn = www.google.cn|www.g.cn  (这块不需要改动）
google_hk = www.google.com|1.1.1.1|www.google.com.hk|2.2.2.2|3.1.1.1|www.l.google.com|www2.l.google.com|www3.l.google.com|www4.l.google.com|20.1.1.1（这个只是例子，配置的时候是域名和好用的IP地址混着写）

[autorange]
threads = 4 （这块调成4-6个即可）

[dns]
enable = 1
servers = dns IP地址使用最好全使用国外IP。使用作者默认的也可以
