class VerifiedHTTPSConnection(httplib.HTTPSConnection):
    def connect(self):
        # overrides the version in httplib so that we do certificate verification
        sock = socket.create_connection((self.host, self.port), self.timeout) #这个socket.create_connection()已经被proxy.py重新定义了
        if self._tunnel_host:
            self.sock = sock
            self._tunnel()
        # wrap the socket using verification with the root certs in trusted_root_certs
        CA_cert_File = os.path.join(os.path.dirname(__file__), 'trusted_root_certs')
        if os.path.exists(CA_cert_File):
            self.sock = ssl.wrap_socket(sock, cert_reqs=ssl.CERT_REQUIRED, ca_certs="trusted_root_certs") #cert_reqs=ssl.CERT_NONE就是不做validation
        else:
            sys.stderr.write('\n\n\ncan not find trusted_root_certs\n')

class VerifiedHTTPSHandler(urllib2.HTTPSHandler):
    def __init__(self, connection_class = VerifiedHTTPSConnection):
        self.specialized_conn_class = connection_class
        urllib2.HTTPSHandler.__init__(self)
    def https_open(self, req):
        return self.do_open(self.specialized_conn_class, req)

#6 xiayujia...@gmail.com 
每次运行proxy.py时，会在local/google_certs目录里生成google的证书。随便找个证书管理工具，比如xca，打开google证书的证书，根据证书的签名CA的名字，下载签名CA的证书，或者从firefox里导出相关CA的证书，然后把那些CA证书粘到一个文本文件里，命令为trusted_root_certs，放在local目录下。就行了。

    def setup(self):

                        common.GOOGLE_HOSTS = tuple(set(sum((tuple(x[-1][0] for x in socket.getaddrinfo(host, 80)) if host[-1] not in '1234567890' else (host,) for host in common.GOOGLE_HOSTS), ())))

        for hostip in common.GOOGLE_HOSTS:

            if google_cert == None:
                sys.stdout.write('can not retrieve google cert\n')
            google_crtFile = os.path.join(os.path.dirname(__file__), 'google_certs', hostip+'.crt')
