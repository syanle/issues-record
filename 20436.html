# 我属于小众，但要说一句，和系统真没啥关系

# 我讨厌写程序，能够不写就不写，能够少写就少写
# 又是c又是python的有些累


cat > goagent_ip.sh << EOF
#!/bin/bash

ini=$1

chk_cert()
{
	curl --connect-timeout 10 --resolve google.com:443:$1 https://google.com >/dev/null 2>&1
	[ $? = 0 ] && echo $1
}
export -f chk_cert

awk -F"= " '/^google_hk/ {print $2}' $ini | sed -e 's/|/\n/g' | parallel -j50 chk_cert {} | tee >(paste -sd\| > /tmp/goagent_ip)
sed -i "s/^google_hk.*$/google_hk = $(cat /tmp/goagent_ip)\r/" $ini
EOF

chmod +x goagent_ip.sh
goagent_ip.sh proxy.user.ini
