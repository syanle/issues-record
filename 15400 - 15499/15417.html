#34 prc...@gmail.com 
#33，你需要手动创建这个文件，然后把跟proxy.ini设置不同的地方填进去，比如：
[gae]
appid = 你的appid
password = 你的密码
window = 1
obfuscate = 1
pagespeed = 1
validate = 1
options = rc4

[iplist]
google_hk = 你用的IP地址
