#16 playne...@gmail.com 
修改local/proxylib.py文件内容
class DirectFetchPlugin(BaseFetchPlugin):
    """direct fetch plugin"""
    connect_timeout = 4 (默认是4，改成8即可，只修改这一处即可）
    read_timeout = 16
（注：此处只限自定义iplist模式需要修改这块,小于4会使找出来的iplist地址马上挂掉）
