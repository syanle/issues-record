#2 mono...@gmail.com 
这个脚本也太简陋了，服务脚本不能这么干啊。
实际上Goagent自带了服务脚本proxy.sh，很方便的。
首先在/etc/init.d/目录下建一个proxy.sh的符号链接，假设goagent存放在用户zhang3的主目录下：
sudo ln -s /home/zhang3/goagent/local/proxy.sh /etc/init.d/goagent
现在可以测试一下服务能否正常启动：
sudo /etc/init.d/goagent start
打开浏览器访问墙外网站测试能否正常打开，如果能正常打开，可以进入下一步，如果不能，查看goagent的运行日志：
cat /var/log/goagent.log
检查错误发生的原因。修正错误后重启goagent服务再次测试，命令如下：
sudo /etc/init.d/goagent restart
确定服务能正常运行后，运行如下命令，使goagent服务随机启动：
sudo update-rc.d goagent defaults
btw:goagent的服务脚本有三个运行参数, start stop restart,平时也可以直接调用脚本控制goagent服务：
sudo /etc/init.d/goagent start|stop|restart

#3 Monistra...@gmail.com 
Try:

############################# file begins #################################
#!/bin/bash
### BEGIN INIT INFO
# Provides:          goagent
# Required-Start:    $remote_fs $syslog $network
# Required-Stop:     $remote_fs $syslog
# X-Stop-After:
# Default-Start:     2 3 4 5
# Default-Stop:      0 1 6
# Short-Description: 
# Description:       
### END INIT INFO

user=nobody
group=nogroup
chuid=nobody
pid=/var/run/goagent.pid
name=goagent
python=/opt/python2.7.8/bin/python
logfile=/tmp/goagent.log
prog=/opt/goagent/local/proxy.py

# Get lsb functions
. /lib/lsb/init-functions

case $1 in
    start)
	log_begin_msg "Starting goagent..."
	start-stop-daemon --start --quiet --oknodo \
		--user "$user" --name "$name" --pidfile \
		"$pid" --make-pidfile --chuid "$chuid"  \
		--startas "$python" -- "$prog" >/dev/null 2>$logfile &
	log_end_msg $?
	;;
    stop)
	log_begin_msg "Stopping goagent..."
	start-stop-daemon --stop --pidfile $pid >/dev/null 2>&1 || true
	log_end_msg $?
	rm -rf $pid $logfile
        ;;
    restart)
	$0 stop
	sleep 1
	$0 start
        ;;
   status)
	status_of_proc -p $pid "goagent" "goagent"
	;;
   *)
	log_success_msg "Usage: service goagent {start|stop|restart}"
    ;;
esac


##################### file end ################################################
modify "python=," "prog=" to your environment, and save to file: goagent

chmod +x goagent
sudo cp goagent /etc/init.d/
sudo service goagent start|stop|restart

sudo update-rc.d goagent defaults
