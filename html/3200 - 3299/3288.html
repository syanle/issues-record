#3 phus.lu 
请到master分支下载，如下编辑[pac]段落
[pac]
enable = 1
ip = 127.0.0.1
port = 8089
file = goagent.pac
update = 1
remote = http://ftp.apnic.net/apnic/stats/apnic/delegated-apnic-latest
timeout = 16
direct = .253874.com|.cnn.com|eweiqi.com

启动，访问http://127.0.0.1:8089/goagent.pac 成功后，再把update = 0，重启goagent.exe
