#1 mql...@gmail.com 
自动退出不用怕，一直检查进程就行了，终止了就重新运行

下面是我一直用的一个脚本，保存到文件，开机自动运行就行
里面的/q是我保存在根目录的一个文件，里面写了 python /local/proxy.py


CheckProcess()
{
  
  if [ "$1" = "python" ];
  then
    return 1
  fi
 
  PROCESS_NUM=`ps -ef | grep "$1" | grep -v "grep" | wc -l` 
  if [ $PROCESS_NUM -eq 1 ];
  then
    return 0
  else
    return 1
  fi
}
 
 
while [ 1 ] ; do
 CheckProcess "/q"
 CheckQQ_RET=$?
 if [ $CheckQQ_RET -eq 1 ];
 then
 
 
  killall  /q
  killall  python
  exec /q &  
 fi
 sleep 1
done
